# Python dependencies for InnoCORESDL. Requires Python >= 3.12.
#   conda activate sdl && pip install -r requirements.txt
#
# Fully self-contained: ALL hardware drivers are vendored in-repo under
# vendor/ (sy01b, entris_ii, mks_motor, lmc all as packages), so
# a fresh `pip install -r requirements.txt` needs no GitHub access or sibling
# clones. Only the drivers' runtime deps (pyserial, pyftdi) are pulled from
# PyPI below. See
# vendor/VENDORED.md for each package's upstream source + commit; update a
# driver by re-copying its source into vendor/ and bumping VENDORED.md.

# ── Transports / libs the vendored drivers + cell need ─────────────────────
pyserial>=3.5    # pump (sy01b), balance (entris_ii), linear (lmc) over serial
pyftdi>=0.55     # XZ gantry (mks_motor) over FTDI USB2CAN

# ── L1 /v1 server (server/) ────────────────────────────────────────────────
fastapi>=0.115        # /v1 app + pydantic schemas
uvicorn>=0.30         # ASGI server (python -m server)

# ── Demo scenario (demo_scenario/) only ────────────────────────────────────
# Runtime deps of the two demo-only vendored drivers; not needed to run the
# cells or the /v1 server.
ika>=2.0              # IKA RCT digital hotplate (vendor/HotplateController)
python-kasa>=0.7      # Tapo P110M smart plug (vendor/SmartPlugController)
